Create TenantController extending ApiController:
- Constructor: inject PlatformCredentialValidator
- index(): $tenants = auth()->user()->tenants()->paginate(20); return $this->success($tenants);
- store(CreateTenantRequest $request):
  - Validate credentials: $isValid = $this->credentialValidator->validate($request->platform_type, $request->api_credentials, $request->platform_url)
  - If invalid: return $this->error([[field: 'api_credentials', message: 'Invalid API credentials', platform_response: $this->credentialValidator->getLastError()]], 422)
  - $tenant = Tenant::create($request->validated())
  - Attach user: auth()->user()->tenants()->attach($tenant->id, [role: 'admin', joined_at: now()])
  - return $this->created(new TenantResource($tenant), [message: 'Tenant created successfully'])
- show(string $id): $tenant = auth()->user()->tenants()->findOrFail($id); return $this->success(new TenantResource($tenant));
- update(UpdateTenantRequest $request, string $id): $tenant = auth()->user()->tenants()->findOrFail($id); $tenant->update($request->validated()); return $this->success(new TenantResource($tenant));
- destroy(string $id): $tenant = auth()->user()->tenants()->findOrFail($id); $tenant->delete(); return $this->noContent();
