```php
public function statusByTenant(Request $request, string $tenantId): JsonResponse
{
    // Validate tenant ownership (prevent cross-tenant enumeration)
    $tenant = auth()->user()->tenants()->where('id', $tenantId)->firstOrFail();

    // Fetch latest sync log for this tenant
    $latestSync = SyncLog::where('tenant_id', $tenantId)
        ->orderBy('created_at', 'desc')
        ->first();

    // Return sync status or null (graceful degradation)
    return response()->json([
        'data' => $latestSync ? SyncLogResource::make($latestSync)->toArray(request()) : null
    ]);
}
```
