  <action>
    Create sync controller for async job dispatch:

    1. Create app/Http/Controllers/Api/V1/SyncController.php:
       ```php
       <?php

       namespace App\Http\Controllers\Api\V1;

       use App\Http\Controllers\ApiController;
       use App\Jobs\ExampleSyncJob;
       use Illuminate\Http\Request;
       use Illuminate\Http\JsonResponse;
       use App\Services\QueueJobTracker;

       class SyncController extends ApiController
       {
           public function dispatch(Request $request): JsonResponse
           {
               $request->validate([
                   'tenant_id' => 'required|uuid|exists:tenants,id',
                   'data' => 'sometimes|array',
               ]);

               $tenantId = $request->input('tenant_id');
               $data = $request->input('data', []);

               // Dispatch job to Redis queue (non-blocking)
               $job = new ExampleSyncJob($tenantId, $data);
               dispatch($job);

               // Track job status
               $jobStatus = app(QueueJobTracker::class)->track(
                   $job,
                   ExampleSyncJob::class
               );

               return $this->created([
                   'job_id' => $jobStatus->job_id,
                       'status' => 'pending',
                       'message' => 'Sync job dispatched successfully',
                   ]);
               }
           }
       }
       ```

    2. Add route in routes/api.php:
       ```php
       Route::middleware('auth:sanctum')->group(function () {
           // ... existing routes ...

           Route::prefix('sync')->group(function () {
               Route::post('/dispatch', [SyncController::class, 'dispatch']);
           });
       });
       ```

    Implementation notes:
    - Returns 202 Accepted (job queued, not completed)
    - Includes job_id for status tracking (Phase 5+ UI will use this)
    - Validates tenant_id exists in tenants table
    - Optional data array for job parameters
    - Uses QueueJobTracker to create JobStatus record
    - Non-blocking: returns immediately after dispatch

    Reference Phase 2 API patterns (ApiController, response format).

  <action>
    Create integration test verifying end-to-end async sync workflow:

    1. Create tests/Feature/Queue/AsyncSyncOperationTest.php:
       ```php
       <?php

       namespace Tests\Feature\Queue;

       use App\Models\User;
       use App\Models\Tenant;
       use Illuminate\Support\Facades\Queue;
       use App\Jobs\ExampleSyncJob;
       use Illuminate\Support\Facades\Event;
       use Illuminate\Queue\Events\JobProcessed;
       use Tests\TestCase;

       class AsyncSyncOperationTest extends TestCase
       {
           private User $user;

           private Tenant $tenant;

           protected function setUp(): void
           {
               parent::setUp();
               $this->user = User::factory()->create();
               $this->tenant = Tenant::factory()->create();
           }

           public function test_sync_dispatches_job_and_returns_immediately(): void
           {
               Queue::fake();

               $response = $this->actingAs($user)
                   ->postJson('/api/v1/sync/dispatch', [
                       'tenant_id' => $this->tenant->id,
                   ]);

               $response->assertStatus(202)
                   ->assertJsonPath('data.status', 'pending')
                   ->assertJsonStructure([
                       'data' => [
                           'job_id',
                           'status',
                           'message',
                       ],
                   ]);

               Queue::assertPushed(ExampleSyncJob::class, function ($job) {
                   return $job->tenantId === $this->tenant->id;
               });
           }

           public function test_sync_request_returns_quickly_non_blocking(): void
           {
               $startTime = microtime(true);

               $response = $this->actingAs($user)
                   ->postJson('/api/v1/sync/dispatch', [
                       'tenant_id' => $this->tenant->id,
                   ]);

               $duration = (microtime(true) - $startTime) * 1000;

               $response->assertStatus(202);
               $this->assertLessThan(100, $duration, 'Request should return in < 100ms');
           }

           public function test_job_executes_async_with_tenant_context(): void
           {
               $this->withoutExceptionHandling();

               Event::fake([JobProcessed::class]);

               dispatch(new ExampleSyncJob($this->tenant->id));

               $this->assertDatabaseHas('job_statuses', [
                   'tenant_id' => $this->tenant->id,
                   'job_type' => ExampleSyncJob::class,
                   'status' => 'pending',
               ]);

               // Run queue worker
               $this->artisan('queue:work', [
                   '--once' => true,
                   '--queue' => 'sync',
               ])->assertExitCode(0);

               Event::assertDispatched(JobProcessed::class);

               $this->assertDatabaseHas('job_statuses', [
                   'tenant_id' => $this->tenant->id,
                   'status' => 'completed',
               ]);
           }

           public function test_multiple_sync_requests_can_be_dispatched_concurrently(): void
           {
               Queue::fake();

               $tenants = Tenant::factory()->count(5)->create();

               foreach ($tenants as $tenant) {
                   $this->actingAs($user)
                       ->postJson('/api/v1/sync/dispatch', [
                           'tenant_id' => $tenant->id,
                       ])->assertStatus(202);
               }

               Queue::assertPushed(ExampleSyncJob::class, 5);
           }
       }
       ```

    Implementation notes:
    - Queue::fake() verifies job dispatch without processing
    - Response time test proves non-blocking behavior
    - Artisan queue:work test verifies actual async execution
    - Database assertions verify job status tracking
    - Concurrent dispatch test proves queue can handle multiple jobs

    Reference 04-VALIDATION.md test strategy for SYNC-02.

       ```php
       <?php

       namespace Tests\Integration\Jobs;

       use App\Models\Tenant;
       use App\Models\User;
       use App\Jobs\ExampleSyncJob;
       use Illuminate\Support\Facades\DB;
        use Illuminate\Support\Facades\Event;
       use Illuminate\Queue\Events\JobProcessed;
       use Tests\TestCase;

       class TenantContextJobIntegrationTest extends TestCase
       {
           private User $user;

           private Tenant $tenant1;

           private Tenant $tenant2;

           protected function setUp(): void
           {
               parent::setUp();
               $this->user = User::factory()->create();
               $this->tenant1 = Tenant::factory()->create();
               $this->tenant2 = Tenant::factory()->create();
           }

           public function test_job_restores_tenant_context_from_middleware(): void
           {
               Event::fake([JobProcessed::class]);

               dispatch(new ExampleSyncJob($this->tenant1->id));

               $this->artisan('queue:work', [
                   '--once' => true,
                   '--queue' => 'sync',
               ])->assertExitCode(0);

               Event::assertDispatched(JobProcessed::class);

               // Verify logs show correct tenant context
               $logContents = file_get_contents(storage_path('logs/laravel.log'));
               $this->assertStringContainsString('tenant_name: ' . $this->tenant1->name, $logContents);
           }

           public function test_job_queries_respect_tenant_global_scope(): void
           {
               // This test will be enhanced in Phase 6 when we have tenant-scoped models
               // For now, verify Tenant::current() works
               Event::fake([JobProcessed::class]);

               dispatch(new ExampleSyncJob($this->tenant2->id));

               $this->artisan('queue:work', [
                   '--once' => true,
                   '--queue' => 'sync',
               ])->assertExitCode(0);

               // Verify tenant context was set
               $this->assertDatabaseHas('job_statuses', [
                   'tenant_id' => $this->tenant2->id,
                   'status' => 'completed',
               ]);
           }

           public function test_tenant_context_cleared_after_job_execution(): void
           {
               dispatch(new ExampleSyncJob($this->tenant1->id));

               $this->artisan('queue:work', [
                   '--once' => true,
                   '--queue' => 'sync',
               ])->assertExitCode(0);

               // After job completes, tenant context should be null
               $this->assertNull(Tenant::current());
           }

           public function test_failed_job_does_not_leak_tenant_context(): void
           {
               $this->withoutExceptionHandling();

               // Create job that will fail
               $failingJob = new class($this->tenant1->id) extends ExampleSyncJob {
                   public function handle(): void
                   {
                       throw new \Exception('Intentional failure');
                   }
               };

               dispatch($failingJob);

               $this->artisan('queue:work', [
                   '--once' => true,
                   '--queue' => 'sync',
               ])->assertExitCode(0); // Worker exits cleanly

               // Verify failure was tracked
               $this->assertDatabaseHas('job_statuses', [
                   'tenant_id' => $this->tenant1->id,
                   'status' => 'failed',
                   'error_message' => 'Intentional failure',
               ]);

               // Verify context cleared
               $this->assertNull(Tenant::current());
           }
       }
       ```

    Implementation notes:
    - Tests verify SetTenantContext middleware behavior
    - Tenant::current() verified during job execution
    - Global scope testing will be enhanced in Phase 6
