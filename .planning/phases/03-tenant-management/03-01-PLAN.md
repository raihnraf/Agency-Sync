  <action>
Create PHP 8.1 backed enums for PlatformType (shopify, shopware) and TenantStatus (active, pending_setup, sync_error, suspended).

Create tenants table migration:
- uuid('id')->primary()
- string('name')
- string('slug')->unique()
- enum('platform_type', ['shopify', 'shopware'])
- string('platform_url')
- enum('status', ['active', 'pending_setup', 'sync_error', 'suspended'])->default('pending_setup')
- json('api_credentials') - encrypted at app layer via cast
- json('settings')->nullable()
- timestamp('last_sync_at')->nullable()
- string('sync_status')->nullable()
- timestamps()
- softDeletes()

Create tenant_user pivot table:
- id()
- uuid('tenant_id')
- foreignId('user_id')->constrained()->onDelete('cascade')
- string('role')->default('member')
- timestamp('joined_at')->nullable()
- timestamps()
- foreign('tenant_id')->references('id')->on('tenants')->onDelete('cascade')
- unique(['tenant_id', 'user_id'])

Add current_tenant_id to users:
- uuid('current_tenant_id')->nullable()
- foreign('current_tenant_id')->references('id')->on('tenants')->onDelete('set null')
  </action>
